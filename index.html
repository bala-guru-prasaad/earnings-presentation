Risk Assessment Algorithm


// Portfolio Risk Calculator
const portfolioData = { assets: 4.2e9, liabilities: 1.8e9 };

function calculateRiskMetrics(portfolio) {
    const debtToAsset = portfolio.liabilities / portfolio.assets;
    const riskLevel = debtToAsset < 0.5 ? 'Low' : 'Moderate';
    return { ratio: debtToAsset.toFixed(2), level: riskLevel };
}

// Real-time Risk Assessment
const metrics = calculateRiskMetrics(portfolioData);
console.log(`Debt-to-Asset Ratio: ${metrics.ratio}`);  
// Output: 0.43 (Low Risk)
console.log(`Risk Level: ${metrics.level}`);

Assessment: Low Risk Profile (0.43 ratio)
